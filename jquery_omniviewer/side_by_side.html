jQuery OmniViewer (Zoomify flavour)

Interactive Tutorial

To initialize an instance

			var ow1 = $("#ow_1").OmniViewer({
					image: ['VF_0178']
					,server: 'http://images.cch.kcl.ac.uk/dev/ow_zoom_flav'
					,credit: credit
					,zoom: 1
					,render: 'random'
					,showNavButtons: true
					,showNavigation: true
					,debug : true
					,fileFormat: "zoomify"
					,target_tmpl : $.template(null, "
" )
					,target_tmpl_data : [{className:"target",id:"xyz"}]
					,mode : null
					,scrollTo:function(event, data) {alert(data.dx+", "+data.dy);} 
					,ready:function(event,data){console.warn("OW GUI loaded")}
					,pan:function(event,data){
							   console.warn("dx: %i,dy:%i,x:%s,y:%s",data.dx,data.dy,data.absx,data.absy);
						   }
					}).css("width","800px").css("height","800px").css("float","left").css("border","1px solid black").css("margin-right","100px");
