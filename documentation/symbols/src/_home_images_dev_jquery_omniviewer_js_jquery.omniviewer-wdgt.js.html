 64 	_load:function(){
 65 		/*
 66 		* Calls the IIPImage server
 67 		*/
 68 		if(this.fileFormat == "iip"){
 69 			var query_string = "&obj=IIP,1.0&obj=Max-size&obj=Tile-size&obj=Resolution-number";
 70 			var self = this;
 71 			// issue the ajax query
 72 			$.ajax({
 73 			 url: this.server + "?" +"FIF=" + this.images[0].src + query_string,
 74 			 context:self,
 75 			 success: function(data){
 76 				var response = data || alert( "No response from server " + this.server );
 77 				this._info(response);
 78 				var tmp = response.split( "Max-size" );
 79 				if(!tmp[1]) alert( "Unexpected response from server " + this.server );
 80 				var size = tmp[1].split(" ");
 81 				this.max_width = parseInt( size[0].substring(1,size[0].length) );
 82 				this.max_height = parseInt( size[1] );
 83 				tmp = response.split( "Tile-size" );
 84 				size = tmp[1].split(" ");
 85 				this.tileSize[0] = parseInt( size[0].substring(1,size[0].length) );
 86 				this.tileSize[1] = parseInt( size[1] );
 87 				tmp = response.split( "Resolution-number" );
 88 				this.num_resolutions = parseInt( tmp[1].substring(1,tmp[1].length) );
 89 				this.res = this.num_resolutions;
 90 				this._createWindows();
 91 				
 92 			 },
 93 			 error:function(){
 94 				console.error("Unable to get image and tile sizes from server!");
 95 			 },
 96 			});
 97 		}
 98 		/*
 99 		* Calls the Zoomify pseudo-server
100 		*/
101 		else if(this.fileFormat == "zoomify"){
102 			var query_string = "ImageProperties.xml";
103 			var self = this;
104 			$.ajax({
105 			 url: this.server + "/"+this.images[0].src + "/"+ query_string,
106 			 context:self,
107 			 success: function(data){

109 				this.max_height = parseInt(1);
110 				this.tileSize[0] = this.standardTileSize;

122 				this._initialiseZoomify();
123 				this._createWindows();
124 			 },

126 				console.error("Unable to get image and tile sizes from server!");
127 			 },
128 			});

131 		* Calls the Djatoka image server
132 		*/

258 	_initialiseZoomify : function(){
259 		var tiles = [2];
260 		var imageSize =  [2];
261 		tiles[0] = Math.ceil( this.max_width / this.standardTileSize );
262 		tiles[1] = Math.ceil( this.max_height / this.standardTileSize );
263 		imageSize[0] = this.max_width;
264 		imageSize[1] = this.max_height;
265 		
266 		this.tierSizeInTiles.push(new Array(tiles[0],tiles[1]));
267 		this.tierImageSize.push( imageSize );
268 		this._log("Tiles %s",tiles);
269 
270 		while (imageSize[0] > this.standardTileSize ||
271 		       imageSize[1] > this.standardTileSize ) {
272 
273 		    imageSize[0] = Math.floor( imageSize[0]/ 2 );
274 			imageSize[1] = Math.floor( imageSize[1] / 2 );
275 			
276 			tiles[0] = Math.ceil( imageSize[0] / this.standardTileSize );        
277 			tiles[1] = Math.ceil( imageSize[1] / this.standardTileSize );
278 			this._log("pre "+this.tierSizeInTiles);
279 		    this.tierSizeInTiles.push(new Array(tiles[0],tiles[1]));
280 		    this._log("post "+this.tierSizeInTiles);
281 		    this.tierImageSize.push( imageSize );
282 		    this._log(this.tierImageSize);
283 		}
284 
285 		this.tierSizeInTiles.reverse();
286 		this.tierImageSize.reverse();
287 		this.numberOfTiers = this.tierSizeInTiles.length;
288 
289 		this.tileCountUpToTier[0] = 0;      
290 		for (var i = 1; i < this.numberOfTiers; i++) {
291 			//this._log("this.tierSizeInTiles tier=%i %i",i-1,this.tierSizeInTiles[i-1][0]);
292 			var temp = this.tierSizeInTiles[i-1][0] * this.tierSizeInTiles[i-1][1] +
293 		        this.tileCountUpToTier[i-1];
294 		    //this._log("tileCountUpToTier %i",temp);
295 		    this.tileCountUpToTier.push(
296 		        temp
297 		        );
298 		    //this._log("this.tileCountUpToTier tier=%i %i",i,temp);
299 		}
300 		this.num_resolutions = this.numberOfTiers;
301 		this.res = this.num_resolutions;

309 	/**
310 	Calculates the minimum dimensions for the navigation window.
311 	*/
312 	_calculateMinSizes : function(){
313 		var tx = this.max_width;
314 		var ty = this.max_height;
315 		var thumb = 100;
316 	
317 		var winWidth = this.element.width();
318 		var winHeight = this.element.height();
319 		this._log("minwidth=%i minheight=%i",winWidth,winHeight);
320 		if( winWidth>winHeight ){
321 		  // For panoramic images, use a large navigation window
322 		  if( tx > 2*ty ) thumb = winWidth / 2;
323 		  else thumb = winWidth / 4;
324 		}
325 		else thumb = winHeight / 4;
326 	
327 		var r = this.res;
328 		while( tx > thumb ){
329 		  tx = parseInt(tx / 2);
330 		  ty = parseInt(ty / 2);
331 		  // Make sure we don't set our navigation image too small!
332 		  if( --r == 0 ) break;
333 		}
334 		this.min_x = tx;
335 		this.min_y = ty;
336 	
337 		// Determine the resolution for this image view
338 		tx = this.max_width;
339 		ty = this.max_height;
340 		while( tx > winWidth && ty > winHeight ){
341 		  tx = parseInt(tx / 2);
342 		  ty = parseInt(ty / 2);
343 		  this.res--;
344 		}
345 		this.wid = tx;
346 		this.hei = ty;
347 		this.res--;
348 		this._log("CalcMinSizes: wid=%i hei=%i",this.wid,this.hei);
349 		return;

471 	/**
472 	* Zooms in and fires the "zoomIn" event.
473 	*/
474 	_zoomIn : function(){
475 		this._trigger("zoomIn", null);
476    		if( (this.wid <= (this.max_width/2)) && (this.hei <= (this.max_height/2)) ){
477 		   this.res++;
478 		   this.wid = this.max_width;
479 		   this.hei = this.max_height;
480 			if(this.fileFormat == "djatoka"){
481 		  		var iter = this.num_resolutions;
482 			}
483 			else{
484 				var iter = this.num_resolutions-1;
485 			}
486 		   for( var i=this.res; i<iter; i++ ){
487 		 this.wid = Math.floor(this.wid/2);
488 		 this.hei = Math.floor(this.hei/2);
489 		   }
490 	 
491 		   if( this.xfit == 1 ){
492 		 this.rgn_x = this.wid/2 - (this.rgn_w/2);
493 		   }
494 		   else if( this.wid > this.rgn_w ) this.rgn_x = 2*this.rgn_x + this.rgn_w/2;
495 	 
496 		   if( this.rgn_x > this.wid ) this.rgn_x = this.wid - this.rgn_w;
497 		   if( this.rgn_x < 0 ) this.rgn_x = 0;
498 	 
499 		   if( this.yfit == 1 ){
500 		 this.rgn_y = this.hei/2 - (this.rgn_h/2);
501 		   }
502 		   else if( this.hei > this.rgn_h ) this.rgn_y = this.rgn_y*2 + this.rgn_h/2;
503 	 
504 		   if( this.rgn_y > this.hei ) this.rgn_y = this.hei - this.rgn_h;
505 		   if( this.rgn_y < 0 ) this.rgn_y = 0;
506 	 
507 		   this._requestImages();
508 		   if(this.showNavigation){
509 		   	this._positionZone();
510 		   	
511 		   }
512 		   if( this.scale ) this._setScale();
513 	 
514 		 }
515 		 return;

698 			   else{

700 				 tileIndex=parseInt(tileIndex/256);
